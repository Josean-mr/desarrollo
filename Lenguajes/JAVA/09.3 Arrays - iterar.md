# Iterator
```java
package iterar;

import java.util.ArrayList;
import java.util.Iterator;

public class IterarIterator {

public static void main(String[] args) {

// Crear una collección de la clase Coche
ArrayList<Coche> coche = new ArrayList<Coche>();

// Agregar objetos de la clase Coche a la lista
coche.add(new Coche("Seat", 110));
coche.add(new Coche("Citroen", 90));
coche.add(new Coche("BMW", 150));
coche.add(new Coche("Mercedes", 180));

// Crear un iterador para recorrer la lista
Iterator<Coche> it = coche.iterator();

// Recorrer la lista usando el iterador
while (it.hasNext()) {

	/*
	* Obtener el siguiente elemento de la lista y
	* almacenarlo en la variable cocheIterado.
	* La clase Coche representa un vehículo con un nombre y
	* una potencia en CV (caballos de vapor).
	*/
	Coche cocheIterado = it.next();
	
	System.out.println("Coche " + cocheIterado.getNombre());

		}

	}

}

// Clase Coche
class Coche {

private String nombre;
private int potencia;

public Coche(String nombre, int potencia) {
this.nombre = nombre;
this.potencia = potencia;

}  

public String getNombre() {
return nombre;
}

public void setNombre(String nombre) {
this.nombre = nombre;
}

public int getPotencia() {
return potencia;
}

public void setPotencia(int potencia) {
this.potencia = potencia;
}

}
```
